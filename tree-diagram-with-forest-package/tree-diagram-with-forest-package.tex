\documentclass[border=4pt]{standalone}
\usepackage{forest}

\forestset{
  frbox/.style={
    draw=cyan!20,
    fill=#1,
    line width=1pt,
    anchor=north,
    text width=4cm,
    minimum height=5ex
  },
  leftn/.style={
    text width=#1,
    draw=cyan,
    double,
    line width=1pt,
    align=left,
    anchor=west,
    minimum height=5ex
  }
}

                  ]
                ]
              ]
            ]
          ]
        ]
      ]
      [Right child,frbox=cyan!30
      ]
    ]
  ]
]
\draw ([xshift=-3pt]left.south) -- ([shift={(-3pt,-5pt)}]last.south west);
\end{forest}

\end{document}
