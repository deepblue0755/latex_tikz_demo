\node[process,  below=of S12] (S13) {\textbf{Step 9} 重新打包生成新 ISO\\写出 \texttt{.sha256} 校验文件\\(Step 9.5 释放 workdir 锁)};

\node[decision, left=of S16]  (S17) {Step 1$\sim$10\\是否抛出\\异常？};

\node[exitstyle, right=of EXIT0](EXIT1) {结束\\退出码 1};
